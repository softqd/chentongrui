类和对象

// 定义Student类
class Student {
    constructor(name, age, school, grade) {
        this.name = name;
        this.age = age;
        this.school = school;
        this.grade = grade;
    }
    
    learn(subject = "语文") {
        return `${this.name}正在学习${subject}，真棒！`;
    }
    
    exam() {
        return `${this.name}正在考试`;
    }
}

// 定义Undergraduate类继承Student
class Undergraduate extends Student {
    constructor(name, age, school, grade, major, experience) {
        super(name, age, school, grade);
        this.major = major;
        this.experience = experience;
    }
    
    intern(corp) {
        return `${this.name}在${corp}实习，获得经验值1000`;
    }
}

// 使用Promise处理异步
function asyncTask() {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve("异步操作完成！");
        }, 1500);
    });
}

asyncTask().then(result => {
    console.log(result);
});

// 事件处理示例
document.getElementById("eventBtn").addEventListener("click", function() {
    let output = document.getElementById("eventOutput");
    output.innerHTML = "按钮被点击了！";
    
    // 改变按钮样式
    this.style.transform = "scale(0.95)";
    setTimeout(() => {
        this.style.transform = "scale(1)";
    }, 200);
});

// 鼠标事件
let eventBox = document.getElementById("eventOutput");
eventBox.addEventListener("mouseenter", function() {
    this.style.backgroundColor = "#4a69bd";
    this.style.color = "white";
});

eventBox.addEventListener("mouseleave", function() {
    this.style.backgroundColor = "#f1f2f6";
    this.style.color = "#333";
});
