类与对象
class Car { constructor(brand, model, year) { this.brand = brand; this.model = model; this.year = year; } start() { console.log(`汽车 ${this.brand} ${this.model} 启动了`); } stop() { console.log(`汽车 ${this.brand} ${this.model} 停止了`); } } class ElectricCar extends Car { constructor(brand, model, year, range) { super(brand, model, year); this.range = range; } charge() { console.log(`电动汽车 ${this.brand} ${this.model} 正在充电`); } }
